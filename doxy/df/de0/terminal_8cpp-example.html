terminal.cpp

This example shows the extra functions available to terminal operations. These functions, then(), value(), value_or() return a new terminal operation, so they can be combined and chained together like other any terminal. The functions value() and value_or() should only be used when the terminal returns an flow::optional object.

#include <examples.h> // defines vec1to4 = {1,2,3,4}, vec5to8 = {5,6,7,8}
void terminal_example() {
std::cout << "flow::detail::Terminal" << std::endl;
std::cout << "Terminal::then(F func): applies `func` to the result of this terminal operation." << std::endl;
// F is a function that takes the result of this terminal as it's only parameter and returns a new result (or void)
vec1to4 | any([](int i) { return i > 5; }).then(std::logical_not<bool>()); // true
vec5to8 | sum(0).then([](int i) { return i*i; }); // (5+6+7+8)^2 = 26*26 = 676
std::cout << "Terminal::value(): directly access the value of an optional return value." << std::endl;
// if there is no value in the optional, this leads to undefined results
vec5to8 | sum().value(); // 26; no need to use operator* to get the value
vec1to4 | map([](int i) { return 2 * i; }) | max().value(); // 8
std::cout << "Terminal::value_or(T def): access the contained value of an optional, or use `def` if no value exists." << std::endl;
// as per optional::value_or, a copy of the value in the optional is returned
vec1to4 | filter([](int i) { return i > 100; }) | sum().value_or(-1); // -1; stream was empty so no value in sum
vec1to4 | sum().value_or(-1); // (1+2+3+4) = 10; there was a value
std::cout << "Terminal::if_then(F func, G other): applies `func` to the result if the optional has a value, `other` otherwise." << std::endl;
// F is a function that takes the underlying type in the optional, ie optional<T> -> takes a T&; it's result is returned
// G takes no parameters and it's result and returned
vec1to4 | sum().if_then([](int i) { std::cout << "sum = " << i << std::endl; }); // displays "sum = 10"
empty<int>() | sum().if_then([](int i) { std::cout << "sum = " << std::endl; },
[]() { std::cout << "no sum!" << std::endl; }); // displays "no sum!"
vec5to8 | sum().if_then([](int i) { std::cout << i; }, []() { std::cout << "empty"; }); // displays 26
int r = empty<int>() | sum().if_then([](int i) { return i*i; }, []() { return -1; }); // r = -1
r = vec1to4 | sum().if_then([](int i) { return i*i; }, []() { return -1; }); // r = 10*10 = 100
std::cout << std::endl << std::endl;
}
